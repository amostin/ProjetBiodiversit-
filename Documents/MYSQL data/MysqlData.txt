#######################
CREATE TABLES         #
#######################

CREATE TABLE `parcours` (
  `ParcoursID` int NOT NULL AUTO_INCREMENT,
  `ParcoursNom` varchar(45) DEFAULT NULL,
  `ParcoursLongueur` float DEFAULT NULL,
  PRIMARY KEY (`ParcoursID`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4_general_ci;

CREATE TABLE `familles` (
  `FamilleID` int NOT NULL AUTO_INCREMENT,
  `FamilleNom` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`FamilleID`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4_general_ci;

CREATE TABLE `categories` (
  `CategorieID` int NOT NULL AUTO_INCREMENT,
  `CategorieNom` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`CategorieID`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4_general_ci;

CREATE TABLE `pointsinteret` (
  `PointInteretID` int NOT NULL AUTO_INCREMENT,
  `NomScientifique` varchar(45) DEFAULT NULL,
  `Nom` varchar(45) DEFAULT NULL,
  `FamilleID` int DEFAULT NULL,
  `ParcoursID` int DEFAULT NULL,
  `Longitude` float DEFAULT NULL,
  `Latitude` float DEFAULT NULL,
  `CategorieID` int DEFAULT NULL,
  `Accessibilite` tinyint DEFAULT '1',
  `Debut` date DEFAULT NULL,
  `Fin` date DEFAULT NULL,
  PRIMARY KEY (`PointInteretID`),
  KEY `FK_Parcours_ParcoursID_idx` (`ParcoursID`),
  KEY `FK_Categories_CategorieID_idx` (`CategorieID`),
  KEY `FK_Familles_FamilleID_idx` (`FamilleID`),
  CONSTRAINT `FK_Categories_CategorieID` FOREIGN KEY (`CategorieID`) REFERENCES `categories` (`CategorieID`),
  CONSTRAINT `FK_Familles_FamilleID` FOREIGN KEY (`FamilleID`) REFERENCES `familles` (`FamilleID`),
  CONSTRAINT `FK_Parcours_ParcoursID` FOREIGN KEY (`ParcoursID`) REFERENCES `parcours` (`ParcoursID`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

CREATE TABLE `users` (
  `UserID` int NOT NULL AUTO_INCREMENT,
  `Nom` varchar(45) DEFAULT NULL,
  `Pseudo` varchar(45) DEFAULT NULL,
  `MdP` longtext,
  `Admin` tinyint DEFAULT '0',
  PRIMARY KEY (`UserID`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;
